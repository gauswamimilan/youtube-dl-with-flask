server {

  listen 80;
  server_name localhost;
  
  location /download/ {
    internal; 
    root /home/user/python;
  }
  
  location / {
    proxy_set_header X-Sendfile-Type X-Accel-Redirect;
    proxy_set_header Host $http_host;
    proxy_redirect off;
    if (!-f $request_filename) {
        proxy_pass http://127.0.0.1:8000;
        break;
    }
  }
}
